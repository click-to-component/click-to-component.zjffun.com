<StepItem title="Usage">

<Note title="Important" type="danger" className="mb-8">
Do not enable `babel-plugin-transform-react-jsx-source-to-style` in product enviroument.
</Note>

<Tabs defaultValue="babel.config.js" className="pt-5 mb-1">
<TabsList className="">
  <TabsTrigger value="babel.config.js">babel.config.js</TabsTrigger>
</TabsList>
<TabsContent value="babel.config.js">

```diff-js
+const plugins = [];

+if (process.env.NODE_ENV === "development") {
+  plugins.push("babel-plugin-transform-react-jsx-source-to-style");
+}

module.exports = {
  presets: [
    [
      "taro",
      {
        framework: "react",
        ts: true,
        compiler: "webpack5",
      },
    ],
  ],
+  plugins: [process.env.NODE_ENV === "development"],
};
```

Or

```diff-js
module.exports = {
  presets: [
    [
      "taro",
      {
        framework: "react",
        ts: true,
        compiler: "webpack5",
      },
    ],
  ],
+  plugins: [process.env.NODE_ENV === "development" && "babel-plugin-transform-react-jsx-source-to-style"].filter(Boolean),
};
```

</TabsContent>
</Tabs>

<TabsContent value="vite">
[`vite.config.ts`](https://github.com/click-to-component/vue-click-to-component/blob/main/examples/vite/vite.config.ts#L7)

```js
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
+import vueClickToComponent from 'vue-click-to-component/vite-plugin';

// https://vitejs.dev/config/
export default defineConfig({
-  plugins: [vue()],
+  plugins: [vueClickToComponent(), vue()],
})
```

</TabsContent>
<TabsContent value="vue-cli">
[`vue.config.js`](https://github.com/click-to-component/vue-click-to-component/blob/main/examples/vue-cli/vue.config.js#L6-L8)

```js
const { defineConfig } = require("@vue/cli-service");
+const vueClickToComponent = require("vue-click-to-component/vue-cli-plugin");

module.exports = defineConfig({
transpileDependencies: true,
+  chainWebpack: (config) => {
+    vueClickToComponent(config);
+  },
});
```
</TabsContent>
<TabsContent value="webpack">
[`webpack.config.js`](https://github.com/click-to-component/vue-click-to-component/blob/main/examples/webpack/webpack.config.js#L31-L35)

+    {
+        test: /\.vue$/,

  {
      test: /\.vue$/,
      loader: 'vue-loader',
  },
],
},
```

[`package.json`](https://github.com/click-to-component/vue-click-to-component/blob/main/examples/webpack/package.json#L10)

}
```
